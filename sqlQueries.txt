drop table person;
drop table address;
create table person(
    id int primary key auto_increment,
    first_name varchar(30),
    last_name varchar(30)
);

create table address(
    id int primary key auto_increment,
    floor int,
    street varchar(30),
    locality varchar(30),
    society varchar(30),
    pin_code varchar(6),
    city varchar(30),
    state varchar(30),
    foreign key (person_id)
    references person(id)
);
